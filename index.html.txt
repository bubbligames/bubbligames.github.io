<script>
var c = document.getElementById("Canvas");
var ctx = c.getContext("2d");

var x = 300;
var y = 300;
var anglex = 0;
var angley = 0;
var rotatex = 0;
var rotatey = 0;
var i = 0;

ctx.fillStyle = "black";
ctx.fillRect(0, 0, 5000, 5000);
//ctx.fillRect(x, y, 5, 5);

function ticker() {
setInterval(alertFunc, 1000);
}

function checkangle(){


if (anglex == 5){               // send to beginning
anglex = 1;
rotatex = 5;
}
if (angley == 5){
angley = 1;
rotatey = 0;
}
if (anglex == 4){
rotatex = 0;                                
}                                      // 0;-5  up
if (angley == 4){
rotatey = -5;
}
if (anglex == 3){
rotatex = 5;
}                                    // 5;0    right
if (angley == 3){
rotatey = 0;
}
if (anglex == 2){
rotatex = 0;
}                                  // 0;5  down
if (angley == 2){
rotatey = 5;
}
if (anglex == 1){
rotatex = -5;
}                               // -5;0   left       
if (angley == 1){
rotatey = 0;
}
if (anglex == 0){               //send to top
anglex = 4;
rotatex = 0;   
}                                   
if (angley == 0){
angley = 4;
rotatey = -5;
}  
}

function alertFunc() {

var imgData = ctx.getImageData(x, y, 5, 5);

var red = imgData.data[0];
var green = imgData.data[1];
var blue = imgData.data[2];

i+=1;
document.getElementById("text").innerHTML = "steps:" + i;

if (red == 255){

ctx.fillStyle = "cyan";     // filling (some color combinations work some don't)
ctx.fillRect(x, y, 5, 5);

anglex += -1; 
angley += -1;
checkangle();

x += rotatex; 
y += rotatey;
}

else {

ctx.fillStyle = "red";
ctx.fillRect(x, y, 5, 5);

anglex += 1; 
angley += 1;
checkangle();

x += rotatex; 
y += rotatey;
}
}
